\subsection{Logical Equivalence}
In general, two statements are logical equivalent if their truth values match up line-for-line in a truth table. 
Here we will discuss some special logical equivalent statements.

\subsubsection{DeMorgan's Laws}
Given statements $P$ and $Q$, 
\begin{equation*}
    \neg \left(P \wedge Q\right) = (\neg P \vee \neg Q)
\end{equation*}
\begin{equation*}
    \neg \left(P \vee Q\right) = (\neg P \wedge \neg Q)
\end{equation*}

\subsubsection{Contrapositive Law}
\[
    P \implies Q = (\neg P) \implies (\neg Q)
\]

\subsubsection{Commutative Laws}
\[
    P \vee Q = Q \vee P
\]
\[
    P \wedge Q = Q \wedge P
\]

\subsubsection{Distributive Laws}
\[
    P \vee (Q \wedge R) = (P \vee Q) \wedge (P \vee R)
\]
\[
    P \wedge (Q \vee R) = (P \wedge Q) \vee (P \wedge R)
\]

\subsubsection{Associative Laws}
\[
    (P \wedge Q) \wedge R = P \wedge (Q \wedge R)
\]
\[
    (P \vee Q) \vee R = P \vee (Q \vee R)
\]
