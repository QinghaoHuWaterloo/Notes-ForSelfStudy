\subsection{Quantifiers}
\begin{definition}
    [The Universal Quantifier $\forall$]
    The Universal  quantifier is used to assert that a property holds for every element in a given domain. 
\end{definition}


\begin{definition}
    [The Existential Quantifier $\exists$]
    The Existential  quantifier is used to assert that a property holds for at least one element in a given domain. 
\end{definition}
