% \subsection{Subsets}
\begin{definition}
    [Subsets] $A$ is a subset of $B$, written as $A \subset B$ or $A \subseteq B$, if all elements are in $A$ and in $B$. i.e 
    $$(\forall x)x \in A \implies x \in B$$
\end{definition}

\begin{remark}
    Fun fuct, the empty set $\emptyset$ is a subset of all sets, that is, $\emptyset \subseteq B$ for any set B.
\end{remark}
